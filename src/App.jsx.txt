const sb = async (path, opts = {}) => {
  const r = await fetch(`${SUPABASE_URL}/rest/v1/${path}`, {
    headers: { "apikey": SUPABASE_KEY, "Authorization": `Bearer ${SUPABASE_KEY}`, "Content-Type": "application/json", "Prefer": "return=representation", ...opts.headers },
    ...opts
  });
  const text = await r.text();
  try { return JSON.parse(text); } catch { return text; }
};

const authFetch = async (path, body) => {
  const r = await fetch(`${SUPABASE_URL}/auth/v1/${path}`, {
    method: "POST",
    headers: { "apikey": SUPABASE_KEY, "Content-Type": "application/json" },
    body: JSON.stringify(body)
  });
  return r.json();
};

                        <button key={op} onClick={() => setF("nivel_calidad", op)} style={{ flex: 1, padding: "8px", borderRadius: 8, border: `1px solid ${form.nivel_calidad===op?"#d4af37":"#2a2a20"}`, background: form.nivel_calidad===op?"#d4af3715":"transparent", color: form.nivel_calidad===op?"#d4af37":"#555", fontFamily: "DM Sans, sans-serif", fontSize: 11, cursor: "pointer", fontWeight: 700 }}>{op}</button>
                      ))}
                    </div>
                    <textarea value={form.comentarios_tecnicos} onChange={e=>setF("comentarios_tecnicos",e.target.value)} placeholder="Comentarios tecnicos del taller..."
                      style={{ width: "100%", background: "#0d0d0a", border: "1px solid #2a2a20", borderRadius: 8, padding: "10px 14px", color: "#e8e0d0", fontSize: 14, boxSizing: "border-box", fontFamily: "DM Sans, sans-serif", minHeight: 60, resize: "vertical" }} />
                  </SECTION>
                )}
              </div>
            </div>

            <div style={{ display: "flex", justifyContent: "center", marginTop: 20 }}>
              <button onClick={guardarFormulario} style={{ background: "#d4af37", color: "#000", border: "none", borderRadius: 10, padding: "14px 40px", fontFamily: "DM Sans, sans-serif", fontWeight: 700, fontSize: 14, cursor: "pointer", letterSpacing: 1 }}>
                GUARDAR LEVANTAMIENTO
              </button>
            </div>
          </div>
        )}

        {/* ESTILOS VISUALES */}
        {tab === "estilos" && (
          <div>
            <h1 style={{ color: "#d4af37", margin: "0 0 8px", fontSize: 24 }}>Estilos de Cocina</h1>
            <p style={{ color: "#555", margin: "0 0 24px", fontSize: 13, fontFamily: "DM Sans, sans-serif" }}>Selecciona el estilo que mas te gusta para tu cocina</p>
            <div style={{ display: "grid", gridTemplateColumns: "repeat(auto-fill,minmax(240px,1fr))", gap: 16 }}>
              {ESTILOS.map(e => {
                const sel = form.estilo.includes(e.label);
                return (
                  <div key={e.key} onClick={() => setF("estilo", toggle(form.estilo, e.label))}
                    style={{ background: "#0f0f0a", border: `2px solid ${sel?"#d4af37":"#ffffff08"}`, borderRadius: 16, overflow: "hidden", cursor: "pointer", transition: "all .2s", transform: sel?"scale(1.02)":"none" }}>
                    <div style={{ position: "relative", height: 160, overflow: "hidden" }}>
                      <img src={e.img} alt={e.label} style={{ width: "100%", height: "100%", objectFit: "cover", filter: sel?"brightness(1)":"brightness(0.6)", transition: "all .3s" }} />
                      {sel && <div style={{ position: "absolute", top: 10, right: 10, background: "#d4af37", color: "#000", borderRadius: "50%", width: 28, height: 28, display: "flex", alignItems: "center", justifyContent: "center", fontWeight: 900, fontSize: 14 }}>✓</div>}
                    </div>
                    <div style={{ padding: "14px 16px" }}>
                      <div style={{ fontWeight: 700, fontSize: 15, color: sel?"#d4af37":"#e8e0d0", marginBottom: 4 }}>{e.label}</div>
                      <div style={{ fontSize: 12, color: "#555", fontFamily: "DM Sans, sans-serif" }}>{e.desc}</div>
                    </div>
                  </div>
                );
              })}
            </div>
            {form.estilo.length > 0 && (
              <div style={{ marginTop: 20, background: "#0f0f0a", border: "1px solid #d4af3730", borderRadius: 12, padding: 16 }}>
                <div style={{ fontFamily: "DM Sans, sans-serif", fontSize: 13, color: "#aaa" }}>
                  Estilos seleccionados: <b style={{ color: "#d4af37" }}>{form.estilo.join(", ")}</b>
                </div>
              </div>
            )}
          </div>
        )}

        {/* PRESUPUESTO */}
        {tab === "presupuesto" && (
          <div>
            <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center", marginBottom: 24, flexWrap: "wrap", gap: 10 }}>
              <div>
                <h1 style={{ color: "#d4af37", margin: 0, fontSize: 24 }}>Presupuesto Profesional</h1>
                <p style={{ color: "#555", margin: "4px 0 0", fontSize: 13, fontFamily: "DM Sans, sans-serif" }}>Completa los campos para generar el presupuesto</p>
              </div>
              {totalPresupuesto > 0 && (
                <button onClick={imprimirPresupuesto} style={{ background: "#d4af37", color: "#000", border: "none", borderRadius: 8, padding: "10px 20px", fontFamily: "DM Sans, sans-serif", fontWeight: 700, fontSize: 13, cursor: "pointer" }}>
                  Imprimir / PDF
                </button>
              )}
            </div>

            <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 24 }}>
              <div>
                <SECTION title="Que Incluye" icon="✅">
                  <textarea value={form.incluye} onChange={e=>setF("incluye",e.target.value)}
                    placeholder={"Diseno\nFabricacion\nTransporte\nInstalacion\nAjustes\nLimpieza final"}
                    style={{ width: "100%", background: "#0d0d0a", border: "1px solid #2a2a20", borderRadius: 8, padding: "10px 14px", color: "#e8e0d0", fontSize: 14, boxSizing: "border-box", fontFamily: "DM Sans, sans-serif", minHeight: 120, resize: "vertical" }} />
                </SECTION>

                <SECTION title="Que NO Incluye" icon="❌">
                  <textarea value={form.no_incluye} onChange={e=>setF("no_incluye",e.target.value)}
                    placeholder={"Plomeria\nElectricidad\nAlbanileria\nElectrodomesticos\nIluminacion\nCambios despues de aprobado diseno"}
                    style={{ width: "100%", background: "#0d0d0a", border: "1px solid #2a2a20", borderRadius: 8, padding: "10px 14px", color: "#e8e0d0", fontSize: 14, boxSizing: "border-box", fontFamily: "DM Sans, sans-serif", minHeight: 120, resize: "vertical" }} />
                </SECTION>

                <SECTION title="Forma de Pago" icon="💰">
                  <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr 1fr", gap: 10 }}>
                    <INPUT label="% Anticipo" value={form.anticipo} onChange={e=>setF("anticipo",e.target.value)} placeholder="60" type="number" />
                    <INPUT label="% Antes instalacion" value={form.pago_entrega} onChange={e=>setF("pago_entrega",e.target.value)} placeholder="30" type="number" />
                    <INPUT label="% Contra entrega" value={form.pago_final} onChange={e=>setF("pago_final",e.target.value)} placeholder="10" type="number" />
                  </div>
                </SECTION>

                <SECTION title="Tiempo y Garantia" icon="⏱️">
                  <INPUT label="Tiempo de entrega" value={form.tiempo_entrega} onChange={e=>setF("tiempo_entrega",e.target.value)} placeholder="20 a 30 dias habiles" />
                  <INPUT label="Garantia" value={form.garantia} onChange={e=>setF("garantia",e.target.value)} placeholder="6 meses en instalacion y herrajes" />
                </SECTION>
              </div>

              <div>
                <SECTION title="Desglose de Precios" icon="📊">
                  <INPUT label="Fabricacion (MXN)" value={form.precio_fabricacion} onChange={e=>setF("precio_fabricacion",e.target.value)} placeholder="35000" type="number" />
                  <INPUT label="Instalacion (MXN)" value={form.precio_instalacion} onChange={e=>setF("precio_instalacion",e.target.value)} placeholder="8000" type="number" />
                  <INPUT label="Cubierta (MXN)" value={form.precio_cubierta} onChange={e=>setF("precio_cubierta",e.target.value)} placeholder="12000" type="number" />
                  <INPUT label="Herrajes (MXN)" value={form.precio_herrajes} onChange={e=>setF("precio_herrajes",e.target.value)} placeholder="5000" type="number" />
                  <INPUT label="Otros (MXN)" value={form.precio_otros} onChange={e=>setF("precio_otros",e.target.value)} placeholder="0" type="number" />

                  <div style={{ background: "#1a1208", border: "1px solid #d4af3730", borderRadius: 12, padding: 16, marginTop: 8 }}>
                    {[["Fabricacion",form.precio_fabricacion],["Instalacion",form.precio_instalacion],["Cubierta",form.precio_cubierta],["Herrajes",form.precio_herrajes],["Otros",form.precio_otros]].filter(([,v])=>v).map(([l,v],i) => (
                      <div key={i} style={{ display: "flex", justifyContent: "space-between", fontSize: 13, color: "#aaa", marginBottom: 6, fontFamily: "DM Sans, sans-serif" }}>
                        <span>{l}</span><span>${parseFloat(v).toLocaleString("es-MX")} MXN</span>
                      </div>
                    ))}
                    <div style={{ display: "flex", justifyContent: "space-between", fontSize: 18, fontWeight: 900, color: "#d4af37", paddingTop: 10, borderTop: "1px solid #d4af3730", marginTop: 6 }}>
                      <span>TOTAL</span><span>${totalPresupuesto.toLocaleString("es-MX")} MXN</span>
                    </div>
                    {totalPresupuesto > 0 && (
                      <div style={{ marginTop: 12, fontSize: 12, color: "#555", fontFamily: "DM Sans, sans-serif", lineHeight: 1.8 }}>
                        Anticipo {form.anticipo}%: <b style={{color:"#d4af37"}}>${(totalPresupuesto*parseFloat(form.anticipo||0)/100).toLocaleString("es-MX")} MXN</b><br/>
                        Antes instalacion {form.pago_entrega}%: <b style={{color:"#d4af37"}}>${(totalPresupuesto*parseFloat(form.pago_entrega||0)/100).toLocaleString("es-MX")} MXN</b><br/>
                        Contra entrega {form.pago_final}%: <b style={{color:"#d4af37"}}>${(totalPresupuesto*parseFloat(form.pago_final||0)/100).toLocaleString("es-MX")} MXN</b>
                      </div>
                    )}
                  </div>
                </SECTION>
              </div>
            </div>
          </div>
        )}

        {/* LEADS TALLER */}
        {tab === "leads" && role === "taller" && (
          <div>
            <h1 style={{ color: "#d4af37", margin: "0 0 20px" }}>Proyectos Disponibles</h1>
            {proyectos.length === 0 && <div style={{ color: "#555", fontFamily: "DM Sans, sans-serif", fontSize: 13, padding: 20 }}>No hay proyectos asignados aun</div>}
            {proyectos.map((p, i) => (
              <div key={i} onClick={() => setProyectoSel(proyectoSel?.created_at===p.created_at?null:p)}
                style={{ background: "#0f0f0a", border: `1px solid ${proyectoSel?.created_at===p.created_at?"#d4af37":"#ffffff08"}`, borderRadius: 12, padding: "16px 20px", marginBottom: 10, cursor: "pointer" }}>
                <div style={{ display: "flex", justifyContent: "space-between", flexWrap: "wrap", gap: 8 }}>
                  <div>
                    <div style={{ fontWeight: 700 }}>Proyecto #{i+1} - {p.tipo_cocina?.join(", ") || "Cocina"}</div>
                    <div style={{ fontSize: 12, color: "#555", fontFamily: "DM Sans, sans-serif" }}>{p.estilo?.join(", ")} - {p.created_at?.split("T")[0]}</div>
                  </div>
                  <div style={{ fontSize: 11, color: "#00bcd4", fontWeight: 700, fontFamily: "DM Sans, sans-serif" }}>{p.nivel_calidad || "Sin clasificar"}</div>
                </div>
                {proyectoSel?.created_at === p.created_at && (
                  <div style={{ marginTop: 14, paddingTop: 14, borderTop: "1px solid #ffffff08", fontFamily: "DM Sans, sans-serif", fontSize: 13 }}>
                    <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 8, color: "#aaa" }}>
                      {[["Tipo",p.tipo_cocina?.join(", ")],["Estilo",p.estilo?.join(", ")],["Material",p.material?.join(", ")],["Largo",p.largo],["Altura",p.altura],["Cubierta",p.material_cubierta?.join(", ")],["Bisagras",p.bisagras],["Correderas",p.correderas],["Accesorios",p.accesorios?.join(", ")],["Nivel",p.nivel_calidad],["Materiales taller",p.materiales_solicitados],["Herrajes taller",p.herrajes_solicitados]].filter(([,v])=>v).map(([l,v],j) => (
                        <div key={j}><b style={{color:"#d4af37"}}>{l}:</b> {v}</div>
                      ))}
                    </div>
                    {p.comentarios_tecnicos && <div style={{ marginTop: 10, color: "#aaa" }}><b style={{color:"#d4af37"}}>Comentarios tecnicos:</b> {p.comentarios_tecnicos}</div>}
                    <button onClick={e => { e.stopPropagation(); setForm({...FORM_INIT,...p}); setTab("presupuesto"); }}
                      style={{ marginTop: 14, background: "#d4af37", color: "#000", border: "none", borderRadius: 8, padding: "9px 20px", fontFamily: "DM Sans, sans-serif", fontWeight: 700, fontSize: 12, cursor: "pointer" }}>
                      Hacer Cotizacion
                    </button>
                  </div>
                )}
              </div>
            ))}
          </div>
        )}

        {/* IA ASESOR */}
        {tab === "ia" && (
          <div>
            <h1 style={{ color: "#d4af37", margin: "0 0 8px", fontSize: 24 }}>IA Asesora de Diseno</h1>
            <p style={{ color: "#555", margin: "0 0 24px", fontSize: 13, fontFamily: "DM Sans, sans-serif" }}>Basandose en tu formulario, la IA recomienda materiales, colores y accesorios</p>

            <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 24 }}>
              <div style={{ background: "#0f0f0a", border: "1px solid #d4af3720", borderRadius: 16, padding: 20 }}>
                <h3 style={{ color: "#d4af37", margin: "0 0 16px", fontFamily: "DM Sans, sans-serif", fontSize: 14, letterSpacing: 1 }}>RESUMEN DEL PROYECTO</h3>
                {[["Tipo de cocina",form.tipo_cocina?.join(", ")],["Estilo",form.estilo?.join(", ")],["Material",form.material?.join(", ")],["Acabado",form.tipo_acabado?.join(", ")],["Color principal",form.color_principal],["Cubierta",form.material_cubierta?.join(", ")],["Accesorios",form.accesorios?.join(", ")]].map(([l,v],i) => (
                  <div key={i} style={{ display: "flex", gap: 8, marginBottom: 8, fontSize: 13, fontFamily: "DM Sans, sans-serif" }}>
                    <span style={{ color: "#555", minWidth: 110 }}>{l}:</span>
                    <span style={{ color: v?"#e8e0d0":"#2a2a20" }}>{v || "Sin especificar"}</span>
                  </div>
                ))}
                <button onClick={analizarConIA} disabled={aiLoading}
                  style={{ width: "100%", marginTop: 16, background: aiLoading?"#1a1a10":"#d4af37", color: aiLoading?"#555":"#000", border: "none", borderRadius: 10, padding: "13px", fontFamily: "DM Sans, sans-serif", fontWeight: 700, fontSize: 13, cursor: aiLoading?"not-allowed":"pointer", letterSpacing: 1 }}>
                  {aiLoading ? "Analizando proyecto..." : "ANALIZAR CON IA"}
                </button>
              </div>

              <div style={{ background: "#0f0f0a", border: "1px solid #ffffff08", borderRadius: 16, padding: 20, minHeight: 300 }}>
                <h3 style={{ color: "#d4af37", margin: "0 0 16px", fontFamily: "DM Sans, sans-serif", fontSize: 14, letterSpacing: 1 }}>RECOMENDACION DE LA IA</h3>
                {!aiResult && !aiLoading && (
                  <div style={{ textAlign: "center", padding: "40px 20px", color: "#2a2a20", fontFamily: "DM Sans, sans-serif", fontSize: 13 }}>
                    Llena el formulario y presiona Analizar con IA
                  </div>
                )}
                {aiLoading && <div style={{ color: "#d4af37", fontFamily: "DM Sans, sans-serif", fontSize: 13, padding: "20px 0" }}>Analizando especificaciones...</div>}
                {aiResult && <div style={{ fontSize: 13, color: "#e8e0d0", lineHeight: 1.9, whiteSpace: "pre-wrap", fontFamily: "DM Sans, sans-serif" }}>{aiResult}</div>}
              </div>
            </div>
          </div>
        )}

      </div>
    </div>
  );
}
